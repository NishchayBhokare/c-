        private void button1_Click(object sender, EventArgs e)
        {
            if (NameTextBox.Text == "" && Emailid_text_box.Text == "" )
            {
                MessageBox.Show("Please enter all details", "error", MessageBoxButtons.OKCancel, MessageBoxIcon.Information);
                return;
            }
            
            else if (NameTextBox.Text == "")
                {
                    MessageBox.Show("Please Enter your name", "error", MessageBoxButtons.OK, MessageBoxIcon.Information);
                    return;
                }
                else if (Emailid_text_box.Text == "")
                {
                    MessageBox.Show("Please Enter your Email ID", "error", MessageBoxButtons.OK, MessageBoxIcon.Information);
                    return;
                }
                else if (dateTimePicker1.Text == "")
                {
                    MessageBox.Show("Please select date", "error", MessageBoxButtons.OK, MessageBoxIcon.Information);
                    return;
                }

            //validation for email address
            var input_E = Emailid_text_box.Text;
            var hasSymbol1 = new Regex("^\\S+@\\S+\\.\\S+$");
            if (!hasSymbol1.IsMatch(input_E))
            {
                MessageBox.Show("Invalid Email format");
                return;
            }

            //validtion of password

                var input = textbox_password.Text;
                if (input == "")
                {
                    MessageBox.Show("Please enter password");
                    return;
                }

                var hasNumber = new Regex(@"[0-9]+");
                var hasUpperChar = new Regex(@"[A-Z]+");
                var hasLowerChar = new Regex(@"[a-z]+");
                var hasSymbol = new Regex(@"[!@#$%^&*()_+=\[{\]};:<>|./?,-]+");
            
                if (!hasNumber.IsMatch(input))
                {
                    MessageBox.Show("Password should contains at least one numeric value");
                    return;
                }
                else  if(!hasUpperChar.IsMatch(input))
                {
                    MessageBox.Show("Password should contains at least one uppercase letter");
                    return;
                }
              else  if(!hasLowerChar.IsMatch(input))
                {
                    MessageBox.Show("Password should contains at least one lowercase letter");
                    return;
                }
              else  if(!hasSymbol.IsMatch(input))
                {
                    MessageBox.Show("Password should contains at least one symbol");
                    return;
                }
                else
                    MessageBox.Show("Register Successfully", "Success", MessageBoxButtons.OK, MessageBoxIcon.Information);
        }
